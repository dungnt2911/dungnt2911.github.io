**Example:**

```sql
-- Complex query với CTE và Window Functions
WITH MonthlyRevenue AS (
  SELECT 
    DATE_FORMAT(order_date, '%Y-%m') AS month,
    customer_id,
    SUM(total_amount) AS revenue,
    RANK() OVER (
      PARTITION BY DATE_FORMAT(order_date, '%Y-%m')
      ORDER BY SUM(total_amount) DESC
    ) AS revenue_rank
  FROM orders
  WHERE order_date >= DATE_SUB(NOW(), INTERVAL 12 MONTH)
  GROUP BY DATE_FORMAT(order_date, '%Y-%m'), customer_id
)
SELECT 
  m.month,
  c.name AS customer_name,
  m.revenue,
  m.revenue_rank
FROM MonthlyRevenue m
JOIN customers c ON m.customer_id = c.id
WHERE m.revenue_rank <= 10
ORDER BY m.month DESC, m.revenue_rank;
```
